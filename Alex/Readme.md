def helper_env_loop(env):
    policy = deprl.load_baseline(env)
    returns = []
    qpos = []
    env.seed(SEED)
    for ep in range(1000):
        ret = 0
        obs, _ = env.reset()
        for i in range(2000):
            action = policy.noisy_test_step(obs)
            obs, reward, done, _, _ = env.step(action)
            env.mj_render() #uncomment to render GUI and visualize
            ret += reward
            qpos.append(env.sim.data.qpos[1])
            if done:
                break
        returns.append(ret)
    env.close()
    return returns, qpos
